Objetivo 3: Graduados en STEM y Ecosistemas Emprendedores

1. Carga y Preparación

stem <- read_excel("bachelorsSTEM.xlsx") %>%
  pivot_longer(cols = -TIME, names_to = "Year", values_to = "STEM_grad") %>% #Trasponemos la matriz
  slice(-(1:30)) %>%
  rename(Country = TIME) %>%
  mutate(STEM_grad = as.numeric(STEM_grad)) %>% #Convertimos a numérico
  group_by(Country) %>%
  summarise(STEM_grad = mean(STEM_grad, na.rm = TRUE), .groups = "drop") %>% #Calculamos el promedio de graduados STEM (por si hubiese más de un año)
  mutate(iso3 = countrycode(Country, "country.name", "iso3c")) %>%
  drop_na() #Elimina cualquier fila con valores faltantes
gei <- read_excel("GEI_dataset_2006_2016.xlsx") %>%
  group_by(Country) %>%
  summarise(GEI = mean(GEI, na.rm = TRUE), .groups = "drop") %>% #Calcula la media de GEI por país
  mutate(iso3 = countrycode(Country, "country.name", "iso3c")) %>%
  drop_na()

2. Unión y Análisis

df <- inner_join(stem, gei, by = "iso3") #Unimos ambos datasets por código de País

3. Clustering y Validación

#Basándonos en los criterios de Shilouette y wss, el número de clusters elegidos son dos.
scaled <- scale(df[, c("STEM_grad", "GEI")])
fviz_nbclust(scaled, kmeans, method = "silhouette")

scaled <- scale(df[, c("STEM_grad", "GEI")])
fviz_nbclust(scaled, kmeans, method = "wss")

km <- kmeans(scaled, centers = 3, nstart = 25) #Aplicamos el algoritmo k-medias para dividir los países en 3 clusters.
                                                #Usamos k-medias porque tenemos datos numéricos, escalados, con pocos outliers
fviz_cluster(km, data = scaled)

# Añadir la asignación de clusters al dataframe original
df$cluster <- km$cluster

# Cargar el mapa mundial y unir con datos
world_map <- map_data("world") %>%

mapa_cluster <- left_join(world_map, df, by = "iso3")
